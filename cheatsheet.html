    return a + b;

Python: Loops

                
                # For loop
for i in range(5):
    print(i)

# While loop
n = 5
while n > 0:
    print(n)
    n -= 1

# Nested loops
for i in range(3):
    for j in range(2):
        print(i, j)

Java: Loops

                
// For loop
for (int i = 0; i < 5; i++) {
    System.out.println(i);
}

// While loop
int n = 5;
while (n > 0) {
    System.out.println(n);
    n--;
}

// Nested loops
for (int i = 0; i < 3; i++) {
    for (int j = 0; j < 2; j++) {
        System.out.println(i + ", " + j);
    }
}

Python: Classes and Objects

                
                
class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def greet(self):
        return f"Hello, my name is {self.name} and I am {self.age} years old."

person = Person("Alice", 30)
print(person.greet())

Java: Classes and Objects

                
                
class Person {
    String name;
    int age;

    public Person(String name, int age) {
        this.name = name;
        this.age = age;
    }

    public String greet() {
        return "Hello, my name is " + name + " and I am " + age + " years old.";
    }
}

Person person = new Person("Alice", 30);
System.out.println(person.greet());

        Scanner scanner = new Scanner(System.in);

Python: Methods

                
                # Static Method
class Math:
    @staticmethod
    def square(x):
        return x * x

# Instance Method
class Calculator:
    def add(self, a, b):
        return a + b

print(Math.square(5))  # Output: 25
calc = Calculator()
print(calc.add(2, 3))  # Output: 5

Java: Methods

                
                // Static Method
class Math {
    public static int square(int x) {
        return x * x;
    }
}

// Instance Method
class Calculator {
    public int add(int a, int b) {
        return a + b;
    }
}

public class Main {
    public static void main(String[] args) {
        System.out.println(Math.square(5));  // Output: 25
        Calculator calc = new Calculator();
        System.out.println(calc.add(2, 3));  // Output: 5
    }
}
